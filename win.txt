Set-PSReadLineKeyHandler -Key Ctrl+f -ScriptBlock {
    $dirs = @(
        "$HOME",
        "$HOME\dev",
        "$HOME\.config"
    )

    $selected = $dirs | ForEach-Object {
        Get-ChildItem $_ -Directory -Recurse -Depth 2 -ErrorAction SilentlyContinue
    } | Select-Object -ExpandProperty FullName | fzf

    if ($selected) {
        # Insere o comando 'cd "caminho"; Clear-Host' no buffer
        [Microsoft.PowerShell.PSConsoleReadLine]::Insert("cd `"$selected`"; Clear-Host")
        [Microsoft.PowerShell.PSConsoleReadLine]::AcceptLine()
    }
}
